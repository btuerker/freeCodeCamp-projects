Palindrome Checker

A palindrome is used to verify a sequence of numbers, strings, or letters that are read left to right and right to left to match the same characters or return the same sequence of characters.

  1. Get the strings or numbers from the user.
  2. Take a temporary variable that holds the numbers.
  3. Reverse the given number.
  4. Compare the original number with the reversed number
  5. If the temporary and original number are same, it the number or string is a Palindrome.
  6. Else the given string or number is not the Palindrome.

function palindrome(str) {
  let start = 0;
  let end = str.length - 1;
  while(start < end) {
    if(str[start] !== str[end]) {
      return false;
    }
    start++;
    end--;
  }

  return true;
}

function rot13(str) {
  let result = "";
  for(let i = 0; i < str.length; i++) {
    let current = str[i];
    let code = current.charCodeAt(0);
    if(code < 65 || code > 90) {
      result += String.fromCharCode(code);
      continue;
    }
    let newCode = code - 13;
    if(newCode < 65) {
      newCode = 90 - (13 - (code - 65 + 1));
    }
    result += String.fromCharCode(newCode);
  }
  return result;
}
